def spiral_xy(i, spiral_num):
    """
    Create the data for a spiral.
    """
     = math.pi*i/(16)

    x = (r * math.cos() * spiral_num)/13 + 0.5
    y = (r * math.sin() * spiral_num)/13 + 0.5
